4M1

For the model definition below, simulate observed y values from the prior (not the posterior). \[y_i \sim Normal(\mu,\sigma) \\ \mu \sim Normal(0, 10) \\ \sigma \sim Exponential(1)\]

library(ggplot2)

samples_N <- 1e4
mu <- rnorm(samples_N, 0, 10)
sigma <- rexp(samples_N, 1)
y_sim <- rnorm(samples_N, mu, sigma)

ggplot(data=NULL, aes(x=y_sim)) + 
  geom_histogram(bins=50) + 
  xlab("Simulated Y")

quap_formula <- alist(
  y ~ dnorm(mu, sigma),
  mu ~ dnorm(0,10),
